  raid_queue_id bigint unsigned NOT NULL auto_increment,
  raider bigint  unsigned NOT NULL,
  dungeon_type mediumint(6) unsigned NOT NULL default 0,
  PRIMARY KEY (raid_queue_id)
);
CREATE TABLE Arena_Queue (
  arena_queue_id bigint unsigned NOT NULL auto_increment,
  contestant mediumint(6) unsigned NOT NULL,
  contestant_team bigint unsigned NOT NULL,
  PRIMARY KEY (arena_queue_id),
  FOREIGN KEY (contestant) references Player(player_id),
  FOREIGN KEY (contestant_team) references Party(party_id)
);
